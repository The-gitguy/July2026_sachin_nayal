Day 16 — Advanced CloudFormation & Multi-Region Deployments
🔹 Ansible — Create and Start EC2 Instances
- name: Create EC2 instances
  hosts: localhost
  become: yes
  tasks:
    - name: Launch EC2 instance in us-east-1
      ec2_instance:
        name: sachin-ec2-us-east-1
        key_name: sachin-key
        region: us-east-1
        instance_type: t2.micro
        image: ami-0abcd1234abcd1234
        wait: yes
    - name: Launch EC2 instance in us-west-1
      ec2_instance:
        name: sachin-ec2-us-west-1
        key_name: sachin-key
        region: us-west-1
        instance_type: t2.micro
        image: ami-0abcd1234abcd1234
        wait: yes

🔹 Kubernetes — Multi-Cluster Deployment with Federation
apiVersion: apps/v1
kind: Deployment
metadata:
  name: sachin-deployment
  labels:
    app: sachin-app
spec:
  replicas: 2
  selector:
    matchLabels:
      app: sachin-app
  template:
    metadata:
      labels:
        app: sachin-app
    spec:
      containers:
        - name: nginx
          image: nginx
          ports:
            - containerPort: 80

---

apiVersion: v1
kind: Service
metadata:
  name: sachin-service
spec:
  selector:
    app: sachin-app
  ports:
    - protocol: TCP
      port: 80
🔹 Jenkinsfile — Deploy to Multi-Cluster Environments
pipeline {
    agent any
    stages {
        stage('Build') {
            steps {
                echo 'Building Docker image...'
                docker.build('sachin-web-app')
            }
        }
        stage('Test') {
            steps {
                echo 'Running tests...'
            }
        }
        stage('Deploy to Kubernetes') {
            steps {
                script {
                    if (env.CLUSTER == 'us-east') {
                        sh 'kubectl --context=us-east apply -f kubernetes/deployment.yaml'
                    } else {
                        sh 'kubectl --context=us-west apply -f kubernetes/deployment.yaml'
                    }
                }
            }
        }
    }
}
🔹 GitLab CI/CD — Multi-Region Deployment with Kubernetes
stages:
  - build
  - push
  - deploy

build:
  stage: build
  script:
    - docker build -t sachin-web-app .

push:
  stage: push
  script:
    - docker login -u "$CI_REGISTRY_USER" -p "$CI_REGISTRY_PASSWORD"
    - docker push sachin-web-app

deploy:
  stage: deploy
  script:
    - |
      if [ "$CI_COMMIT_REF_NAME" == "main" ]; then
        kubectl --context=us-east apply -f kubernetes/prod-deployment.yaml
      else
        kubectl --context=us-west apply -f kubernetes/dev-deployment.yaml
      fi
